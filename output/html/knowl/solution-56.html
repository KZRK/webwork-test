  1. \(\displaystyle \begin{aligned} \frac{15x-5}{8x-4} \div \frac{10x+25}{6x-3} &= \frac{15x-5}{8x-4} \cdot \frac{6x-3}{10x+25} \\ &= \frac{5(3x-1)}{4(2x-1)} \cdot \frac{3(2x-1)}{5(2x+1)} \\ &= \frac{5(3x-1)3(2x-1)}{4(2x-1)5(2x+1)} \\ &= \frac{3(3x-1)}{4(2x+1)}, x \neq \frac{1}{2} \end{aligned}\)

  2. \(\displaystyle \begin{aligned} \frac{4}{x^2+3x+2} + \frac{5}{x+2} &= \frac{4}{(x+2)(x+1)} + \frac{5}{x+2} \\ &= \frac{4}{(x+2)(x+1)} + \frac{5(x+1)}{(x+2)(x+1)} \\ &= \frac{4 +5(x+1)}{(x+2)(x+1)} \\ &= \frac{4 +5x+5}{(x+2)(x+1)} \\ &= \frac{5x+9}{(x+2)(x+1)} \end{aligned}\)
